1 /**
  2 * Container de requests AJAX
  3 * @name RQ
  4 * @class
  5 */
  6 var RQ = {
  7     container: [],
  8     /**
  9     * Função para excluir todas as requests do container
 10     * @function
 11     */
 12     killAll: function() {
 13        for(var i in RQ.container) {
 14            if(RQ.container[i]["request"] !== false)
 15                RQ.container[i]["request"].abort();
 16        }
 17        RQ.container = [];
 18     },
 19     /**
 20     * Função para adicionar uma request ao container
 21     * @param req  Request a ser adicionada
 22     * @param req_id  Um id único para a request
 23     * @function
 24     */
 25     add: function(req,req_id) {
 26         req_id = req_id || "rq_"+RQ.container.length + 1;
 27         if(req !== false)
 28             RQ.container.push({id: req_id,request: req});
 29     },
 30     /**
 31     * Função para buscar uma request no container
 32     * @param condition  Recebe a request ou o id da request como parâmetro de busca
 33     * @function
 34     */
 35     find: function(condition) {
 36         var index = -1;
 37         var field = "";
 38         field = (condition["request"] !== undefined ? "request" : "id");
 39         for(var i in RQ.container) {
 40            if(RQ.container[i][field] === condition[field]) {
 41                 index = i;
 42                 break;
 43            }
 44         }
 45         return index;
 46     },
 47     /**
 48     * Função para remover uma request do container
 49     * @param condition  Recebe a request ou o id da request como parâmetro de busca
 50     * @function
 51     */
 52     remove: function(condition) {
 53         var index = RQ.find(condition);
 54         if(index > -1)
 55             RQ.container.splice(index,1);
 56     },
 57     /**
 58     * Função para excluir uma request do container(ela será automaticamente abortada)
 59     * @param condition  Recebe a request ou o id da request como parâmetro de busca
 60     * @function
 61     */
 62     kill: function(condition) {
 63         var index = RQ.find(condition);

 68         }
 69     },
 70     /**
 71     * Função para exibir todas as requests no container
 72     * @function
 73     */
 74     showAll: function() {
 75         var requests = "";
 76         for(var i in RQ.container) {
 77             requests += "id: "+RQ.container[i]["id"]+",request:"+
 78             RQ.container[i]["request"]+"\n";

 81     }
 82 };
